# SP-3 · Base de Conhecimento B2B — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Construir a base B2B (4 tabelas Supabase) + parsers + mapeador `TIM_v1` + importador Python que engole a exportação real da TIM e normaliza num registro canônico por CNPJ.

**Architecture:** Lógica de parsing e mapeamento em módulos Python puros e testáveis (`scripts/b2b_*.py`); o I/O (xlsx + Supabase) isolado no importador. O schema SQL fica num arquivo aplicado via migration. As funções puras são testadas com pytest antes de qualquer toque no banco.

**Tech Stack:** Python 3 + openpyxl (ler xlsx) + httpx (PostgREST) + pytest (testes). Supabase (Postgres/PostgREST). Migration via MCP supabase.

## Global Constraints

- Formato de telefone canônico: `55` + DDD + número (ex: `5547992082320`).
- Qualidade de telefone preservada: `alta` / `media` / `baixa`.
- Os 7 status TIM mantidos VERBATIM: `Não abordado`, `Não consegui contato`, `Sem perfil`, `Sem interesse`, `Já é da TIM`, `Contrato com concorrência`, `Reunião marcada`. Default quando vazio: `Não abordado`.
- Chave canônica da empresa: `cnpj` (text, 14 dígitos, só números).
- Captura B2B; a ponte PF↔PJ é `fenice_b2b_socios.contato_b2c_id` (FK → `fenice_tim_contatos`, nullable).
- Data no formato origem `dd/mm/aaaa` → armazenar ISO `aaaa-mm-dd`.
- `opcao_mei`/`opcao_simples`: `"SIM"`→true, qualquer outro→false.
- Base agnóstica ao formato: mapeador identificado por `formato="TIM_v1"`.
- Planilha real de referência: `docs/Prospects Ramão.xlsx`, aba `Exportação 28.05.2026`, ~60 empresas.
- Fora de escopo: scoring automático, enriquecimento externo, campanhas PAP, CRM visual, Power BI.

```python
"""Parsers dos campos compostos da exportação B2B (formato TIM_v1)."""
import re

_TEL_RE = re.compile(r"\((\d{2})\)\s*(\d+)\s*\((alta|m[eé]dia|media|baixa)\)", re.I)
_SOCIO_RE = re.compile(r"([^(,][^(]*?)\s*\(([^)]*)\)")


def parse_telefones(bruto):
    if not bruto:
        return []
    out = []
    for ddd, num, qual in _TEL_RE.findall(bruto):
        q = qual.lower().replace("é", "e")
        out.append({"numero": "55" + ddd + num, "qualidade": q})
    return out


def parse_emails(bruto):
    if not bruto:
        return []
    out = []
    for part in bruto.split(","):
        e = part.strip().lower()
        if "@" in e and "." in e:
            out.append(e)
    return out


def parse_socios(bruto):
    if not bruto:
        return []
    out = []
    for nome, qual in _SOCIO_RE.findall(bruto):
        nome = nome.strip()
        if nome:
            out.append({"nome_socio": nome, "qualificacao": qual.strip()})
    return out


def parse_links(bruto):
    if not bruto:
        return []
    return [p.strip() for p in bruto.split(",") if p.strip()]
```

```python
"""Mapeador do formato TIM_v1 → registro canônico fenice_b2b_empresas."""
from datetime import datetime

from b2b_parsers import parse_telefones, parse_emails, parse_socios, parse_links

STATUS_DEFAULT = "Não abordado"


def _so_digitos(v):
    return "".join(c for c in (v or "") if c.isdigit())


def _data_iso(v):
    v = (v or "").strip()
    if not v:
        return None
    try:
        return datetime.strptime(v, "%d/%m/%Y").date().isoformat()
    except ValueError:
        return None


def _sim(v):
    return (v or "").strip().upper() == "SIM"


def _txt(v):
    v = (v if v is not None else "")
    return str(v).strip() or None


def mapear_tim_v1(raw):
    empresa = {
        "cnpj": _so_digitos(raw.get("Cnpj")),
        "razao_social": _txt(raw.get("RazaoSocialNomeEmpresarial")),
        "nome_fantasia": _txt(raw.get("NomeFantasia")),
        "uf": _txt(raw.get("Uf")),
        "municipio": _txt(raw.get("Municipio")),
        "bairro": _txt(raw.get("Bairro")),
        "logradouro": _txt(raw.get("Logradouro")),
        "cep": _so_digitos(raw.get("CEP")) or None,
        "numero": _txt(raw.get("Numero")),
        "data_abertura": _data_iso(raw.get("DataAbertura")),
        "porte_receita": _txt(raw.get("PorteReceita")),
        "porte_estimado": _txt(raw.get("PorteEstimado")),
        "faturamento_estimado": _txt(raw.get("FaturamentoEstimado")),
        "funcionarios_estimado": _txt(raw.get("FuncionariosEstimado")),
        "opcao_mei": _sim(raw.get("OpcaoMei")),
        "opcao_simples": _sim(raw.get("OpcaoSimples")),
        "matriz_filial": _txt(raw.get("MatrizFilial")),
        "cnae_fiscal": _txt(raw.get("CnaeFiscal")),
        "descricao_cnae": _txt(raw.get("DescricaoCnaeFiscal")),
        "emails": parse_emails(raw.get("Emails")),
        "telefones": parse_telefones(raw.get("Telefones")),
        "links": parse_links(raw.get("Links")),
        "status": _txt(raw.get("Status")) or STATUS_DEFAULT,
        "observacao": _txt(raw.get("Observação")),
    }
    socios = parse_socios(raw.get("Socios"))
    return empresa, socios
```

```python
"""Importa exportação B2B (formato TIM_v1) para o Supabase.

Uso: python importar_b2b.py "docs/Prospects Ramão.xlsx"
Env: SUPABASE_URL, SUPABASE_SERVICE_KEY
"""
import os
import sys

import httpx

from b2b_mapeador import mapear_tim_v1

ABA_PADRAO = "Exportação 28.05.2026"


def carregar_xlsx(caminho, aba=ABA_PADRAO):
    import openpyxl
    wb = openpyxl.load_workbook(caminho, read_only=True, data_only=True)
    ws = wb[aba]
    linhas = list(ws.iter_rows(values_only=True))
    if not linhas:
        return []
    hdr = [str(c).strip() if c is not None else "" for c in linhas[0]]
    out = []
    for r in linhas[1:]:
        if all(c is None for c in r):
            continue
        out.append({hdr[i]: r[i] for i in range(len(hdr))})
    return out


def montar_payloads(linhas_raw, parceira, formato):
    empresas, socios, staging = [], [], []
    vistos = set()
    for i, raw in enumerate(linhas_raw):
        empresa, socs = mapear_tim_v1(raw)
        cnpj = empresa["cnpj"]
        st = {"linha_num": i + 1, "dados_raw": raw, "cnpj": cnpj or None,
              "processado": bool(cnpj), "erro": None if cnpj else "sem cnpj"}
        staging.append(st)
        if not cnpj:
            continue
        empresa["fontes"] = [parceira]
        empresas.append(empresa)
        for s in socs:
            chave = (cnpj, s["nome_socio"])
            if chave in vistos:
                continue
            vistos.add(chave)
            socios.append({"cnpj": cnpj, **s})
    return {"empresas": empresas, "socios": socios, "staging": staging}


def _post(client, url, key, tabela, rows, on_conflict=None):
    if not rows:
        return
    params = {"on_conflict": on_conflict} if on_conflict else {}
    hdrs = {"apikey": key, "Authorization": f"Bearer {key}",
            "Content-Type": "application/json",
            "Prefer": "return=minimal,resolution=merge-duplicates"}
    r = client.post(f"{url.rstrip('/')}/rest/v1/{tabela}", params=params,
                    headers=hdrs, json=rows)
    r.raise_for_status()


def importar(caminho, sb_url, sb_key, aba=ABA_PADRAO, parceira="TIM"):
    linhas = carregar_xlsx(caminho, aba)
    p = montar_payloads(linhas, parceira, "TIM_v1")
    with httpx.Client(timeout=60) as c:
        hdrs = {"apikey": sb_key, "Authorization": f"Bearer {sb_key}",
                "Content-Type": "application/json", "Prefer": "return=representation"}
        imp = c.post(f"{sb_url.rstrip('/')}/rest/v1/fenice_b2b_importacoes",
                     headers=hdrs, json={"parceira": parceira, "formato": "TIM_v1",
                     "arquivo_nome": os.path.basename(caminho),
                     "linhas_total": len(linhas), "linhas_ok": len(p["empresas"])})
        imp.raise_for_status()
        imp_id = imp.json()[0]["id"]
        for s in p["staging"]:
            s["importacao_id"] = imp_id
        _post(c, sb_url, sb_key, "fenice_b2b_staging", p["staging"])
        _post(c, sb_url, sb_key, "fenice_b2b_empresas", p["empresas"], on_conflict="cnpj")
        _post(c, sb_url, sb_key, "fenice_b2b_socios", p["socios"], on_conflict="cnpj,nome_socio")
    return {"linhas_total": len(linhas), "linhas_ok": len(p["empresas"])}


if __name__ == "__main__":
    if len(sys.argv) < 2:
        print('Uso: python importar_b2b.py "docs/Prospects Ramão.xlsx"')
        sys.exit(1)
    res = importar(sys.argv[1], os.environ["SUPABASE_URL"],
                   os.environ["SUPABASE_SERVICE_KEY"])
    print(f"✅ {res['linhas_ok']}/{res['linhas_total']} empresas importadas.")
```
